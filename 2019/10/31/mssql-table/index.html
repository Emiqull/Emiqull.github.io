创建表

基本表(默认列值可以为Null)

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型，
… );

指定列不能为null

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型 NOT NULL，
… );

指定列为主键

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型 PRIMARY KEY，
… );

指定列有默认值

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型 DEFAULT (‘默认值’)，
… );

指定列有check约束

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型 CHECK (条件)，–(条件如：age >= 18 AND age <= 65)
… );

指定列自动计算

CREATE TABLE 表名(
列名1 列类型，
列名2 AS (计算公式)， –(列名如实发薪水，计算公式可以是基本薪水+奖金。注AS表示该字段为自动计算字段，不能输入值)
… );

指定列自增

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型 IDENTITY(1，1)，–(指定为自动的列必须是整数类型，且该列的约束是不能为空。IDENTITY的第一个参数是起始值，第二个参数是增量跨度)
… );

指定列为外键(推荐与表结构的定义分开写，不推荐此方法创建外键)

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型，
…，
FOREIGN KEY (列名2) REFERENCES 表名2(表名2的列名2) –(表名的外键是表名2的列名2，其中列名2必须是表名2的主键)
);

指定列为主键且自增

CREATE TABLE 表名(
列名1 列类型，
列名2 列类型 IDENTITY(1，1) PRIMARY KEY，
… );

指定两列为联合主键

CREATE TABLE 表明(
列明1 列类型，
列明 2 列类型，
列明3 列类型 PRIMARY KEY (列明1，列明3)，
…
);

修改表结构

修改表名称

ALTER TABLE表名 TO 新表名;
或EXEC sp_rename ‘表名’，’新表名’;

修改列名称

EXEC sp_rename ‘表名.列名’, ‘新列名’, ‘column’;

修改列类型

ALTER TABLE 表名 ALTER COLUMN 列名 新列类型;

新增列

ALTER TABLE 表名 ADD 列名 列类型;

删除列

ALTER TABLE 表名 DROP COLUMN 列名;

新增有默认值列

ALTER TABLE 表名 ADD 列名 列类型 NOT NULL DEFAULT (默认值);

为指定列添加默认值约束

ALTER TABLE 表名 ADD CONSTRAINT 约束名 DEFAULT ‘默认值’ FOR 列名;

删除列约束

ALTER TABLE 表名 DROP CONSTRAINT 约束名;

修改列属性值为not null

ALTER TABLE 表名 ALTER COLUMN 列名 列类型 NOT NULL;

为指定列添加主键约束

ALTER TABLE 表名 ADD CONSTRAINT 约束名 PRIMARY KEY (列名);

为指定列添加唯一约束

ALTER TABLE 表名 ADD CONSTRAINT 约束名 UNIQUE (列名);

为指定列添加check约束

ALTER TABLE 表名 ADD CONSTRAINT 约束名 CHECK (列名 列条件); –如列名是gender，列条件是(gender = N’男’ or gender = N’女’)

为指定列添加外键约束

ALTER TABLE 表名 ADD CONSTRAINT 约束名 FOREIGN KEY (列名) REFERENCES 表名2(列名2);

为指定列添加聚集索引

CREATE CLUSTEREDINDEX 索引名 ON 表名(列名[ASC|DESC]);

为指定列添加非聚集索引

CREATE NONCLUSTERED INDEX 索引名 ON 表名(列名[ASC|DESC]);

为指定列添加唯一索引

CREATE UNIQUE INDEX 索引名 ON 表名(列名[ASC|DESC]);

修改索引名称

EXEC sp_rename N’表名.索引名’，’新索引名’，N’index’;

修改表中的约束只能是先删除旧约束，再添加新约束

删除表

删除表(清空并移除整张表)

DROP TABLE 表名;

清空表

TRUNCATE清空表

TRUNCATE TABLE 表名;

注：清空表记录、释放表空间、不能激活触发器、仅删除整张表记录

DELETE清空表

DELETE FROM 表名 [WHERE … ];

注：不释放表空间、能激活触发器。如果没有WHERE子句，则清空表记录；如果有WHERE子句，则删除满足过滤条件的表记录

查询表结构

查询表是否存在

SELECT * FROM sysobjects WHERE name = ‘表名’;

查询表结构

SELECT CASE WHEN col.colorder = 1 THEN obj.name
ELSE ‘’
END AS 表名,
col.colorder AS 序号 ,
col.name AS 列名 ,
ISNULL(ep.[value], ‘’) AS 列说明 ,
t.name AS 数据类型 ,
col.length AS 长度 ,
ISNULL(COLUMNPROPERTY(col.id, col.name, ‘Scale’), 0) AS 小数位数 ,
CASE WHEN COLUMNPROPERTY(col.id, col.name, ‘IsIdentity’) = 1 THEN ‘1’
ELSE ‘’
END AS 标识 ,
CASE WHEN EXISTS ( SELECT 1
FROM dbo.sysindexes si
INNER JOIN dbo.sysindexkeys sik ON si.id = sik.id
AND si.indid = sik.indid
INNER JOIN dbo.syscolumns sc ON sc.id = sik.id
AND sc.colid = sik.colid
INNER JOIN dbo.sysobjects so ON so.name = si.name
AND so.xtype = ‘PK’
WHERE sc.id = col.id
AND sc.colid = col.colid ) THEN ‘1’
ELSE ‘’
END AS 主键 ,
CASE WHEN col.isnullable = 1 THEN ‘1’
ELSE ‘’
END AS 允许空 ,
ISNULL(comm.text, ‘’) AS 默认值
FROM dbo.syscolumns col
LEFT JOIN dbo.systypes t ON col.xtype = t.xusertype
inner JOIN dbo.sysobjects obj ON col.id = obj.id
AND obj.xtype = ‘U’
AND obj.status >= 0
LEFT JOIN dbo.syscomments comm ON col.cdefault = comm.id
LEFT JOIN sys.extended_properties ep ON col.id = ep.major_id
AND col.colid = ep.minor_id
AND ep.name = ‘MS_Description’
LEFT JOIN sys.extended_properties epTwo ON obj.id = epTwo.major_id
AND epTwo.minor_id = 0
AND epTwo.name = ‘MS_Description’
WHERE obj.name = ‘表名’
ORDER BY col.colorder ;

查询表列名称和列属性

SELECT C.name AS ColumnName, T.name AS ColumnType
FROM syscolumns C LEFT JOIN systypes T ON C.xusertype = T.xusertype
WHERE C.id = object_id(‘表名’);

查询表外键

SELECT fk.name, fk.object_id, OBJECT_NAME(fk.parent_object_id) as referenceTableName
FROM sys.foreign_keys as fk JOIN sys.objects as obj
ON fk.referenced_object_id = obj.object_id
WHERE obj.name=’表名’;

查询重复记录

SELECT c1.* FROM
(SELECT row_number() OVER(order by [columns…] ) AS rownum,* FROM whctrl [WHERE … ]) c1 JOIN
(SELECT row_number() OVER(order by [columns…] )AS rownum,* FROM whctrl [WHERE … ]) c2
WHERE c1.columns … = c2.columns … AND c1.rownum<>c2.rownum;
